\newcommand{\twocols}[3]{
\hspace{-1.1\parindent}
\begin{minipage}[t][][t]{\textwidth}
\begin{minipage}[t][][t]{0.5\textwidth}
#1
\end{minipage}
\hspace{0.05in}
\begin{minipage}[t][][t]{0.45\textwidth}
#3
\end{minipage} 
\end{minipage}
\noindent
#2
}

\newcommand{\figa}[3]{ 
  \begin{figure}
  \refstepcounter{figure}
  \label{#2}
  \begin{center}
  \includegraphics[scale=#1,angle=0,clip]{#2.pdf} \\
  \end{center}
  {\bf Fig.~\ref{#2} #3}
  \end{figure}
}

  \end{center}

}
